🎬 ANIMACIONES CSS DISPONIBLES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

animate-float            Flotación suave (3s, infinito)
animate-pulse-glow       Brillo pulsante (2s, infinito)
animate-shimmer          Destello deslizante (3s, infinito)
animate-slide-up         Entrada desde abajo (0.6s, ease-out)
animate-bounce-in        Rebote al entrar (0.6s, cubic-bezier)
animate-rotate-slow      Rotación continua (8s, linear)
animate-gradient-shift   Gradiente que cambia (3s, ease)
animate-fade-in          Desvanecimiento (0.5s, ease-in)

Usa con Tailwind:
<div className="animate-float">...</div>
